688  later_stop = (stopH > self.weekly_times[i][2] or (stopH == self.weekly_times[i][2] and stopM >= self.weekly_times[i][3]))

690  if later_stop:

693  self.weekly_times[i][3] = stopM

696  if same_hour_later_minute:
